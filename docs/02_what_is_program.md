```cpp
// –£–º–æ–≤–∞ –∑ –∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–æ—é if-else
if (dronesInStock % 2 == 0) {
    // —Ü–µ–π –±–ª–æ–∫, —è–∫—â–æ –∫—ñ–ª—å–∫—ñ—Å—Ç—å –ø–∞—Ä–Ω–∞
    std::cout << "–ö—ñ–ª—å–∫—ñ—Å—Ç—å –ø–∞—Ä–Ω–∞.";
} else {
    // —Ü–µ–π –±–ª–æ–∫, —è–∫—â–æ –∫—ñ–ª—å–∫—ñ—Å—Ç—å –Ω–µ–ø–∞—Ä–Ω–∞
    std::cout << "–ö—ñ–ª—å–∫—ñ—Å—Ç—å –Ω–µ–ø–∞—Ä–Ω–∞.";
}
```

–î–ª—è –ø–æ–±—É–¥–æ–≤–∏ —É–º–æ–≤ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å—Å—è **–æ–ø–µ—Ä–∞—Ç–æ—Ä–∏ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è** (`>`, `<`, `>=`, `<=`, `==` (—Ä—ñ–≤–Ω—ñ—Å—Ç—å), `!=` (–Ω–µ—Ä—ñ–≤–Ω—ñ—Å—Ç—å)) —Ç–∞ **–ª–æ–≥—ñ—á–Ω—ñ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∏** (`&&` (–Ü), `||` (–ê–ë–û), `!` (–ù–ï)).

### –¶–∏–∫–ª–∏ (Loops)

#### –ü—Ä–∏–∫–ª–∞–¥: –Ü—Å—Ç–æ—Ä—ñ—è –ø—Ä–æ –ì–∞—É—Å—Å–∞ —Ç–∞ —Å–∏–ª–∞ –∞–ª–≥–æ—Ä–∏—Ç–º—ñ–≤

–Ü—Å–Ω—É—î –≤—ñ–¥–æ–º–∞ —ñ—Å—Ç–æ—Ä—ñ—è –ø—Ä–æ –º–æ–ª–æ–¥–æ–≥–æ –ö–∞—Ä–ª–∞ –ì–∞—É—Å—Å–∞, —è–∫–æ–º—É –≤—á–∏—Ç–µ–ª—å –¥–∞–≤ –∑–∞–≤–¥–∞–Ω–Ω—è –¥–æ–¥–∞—Ç–∏ –≤—Å—ñ —á–∏—Å–ª–∞ –≤—ñ–¥ 1 –¥–æ 100. –ì–∞—É—Å—Å –º–∞–π–∂–µ –º–∏—Ç—Ç—î–≤–æ –∑–Ω–∞–π—à–æ–≤ –≤—ñ–¥–ø–æ–≤—ñ–¥—å (5050), –ø–æ–º—ñ—Ç–∏–≤—à–∏, —â–æ —Å—É–º–∞ –ø–∞—Ä —á–∏—Å–µ–ª (1+100, 2+99, ...) –∑–∞–≤–∂–¥–∏ –¥–æ—Ä—ñ–≤–Ω—é—î 101. –¶–µ —ñ–ª—é—Å—Ç—Ä—É—î —Å–∏–ª—É **–∞–ª–≥–æ—Ä–∏—Ç–º—ñ—á–Ω–æ–≥–æ –º–∏—Å–ª–µ–Ω–Ω—è** ‚Äî –∑–∞–º—ñ—Å—Ç—å —Ç–∏—Å—è—á—ñ –æ–ø–µ—Ä–∞—Ü—ñ–π, –º–æ–∂–Ω–∞ –∑–Ω–∞–π—Ç–∏ –µ–ª–µ–≥–∞–Ω—Ç–Ω—ñ—à–µ —Ä—ñ—à–µ–Ω–Ω—è.

–£ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—ñ –º–∏ —á–∞—Å—Ç–æ –ø–æ—á–∏–Ω–∞—î–º–æ –∑ —Ä—ñ—à–µ–Ω–Ω—è "–≤ –ª–æ–±" –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é —Ü–∏–∫–ª—ñ–≤.

#### –¶–∏–∫–ª `while`

![–°—Ö–µ–º–∞ —Ä–æ–±–æ—Ç–∏ —Ü–∏–∫–ª—É while](attachments/02_while.png)

–¶–∏–∫–ª `while` ("–ø–æ–∫–∏") –≤–∏–∫–æ–Ω—É—î —Ç—ñ–ª–æ —Ü–∏–∫–ª—É –¥–æ—Ç–∏, –¥–æ–∫–∏ –π–æ–≥–æ —É–º–æ–≤–∞ —î —ñ—Å—Ç–∏–Ω–Ω–æ—é. –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ —É–º–æ–≤–∏ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è **–ø–µ—Ä–µ–¥** –∫–æ–∂–Ω–æ—é —ñ—Ç–µ—Ä–∞—Ü—ñ—î—é.

**–ö–æ–ª–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏:** –Ü–¥–µ–∞–ª—å–Ω–æ, –∫–æ–ª–∏ –º–∏ –Ω–µ –∑–Ω–∞—î–º–æ —Ç–æ—á–Ω—É –∫—ñ–ª—å–∫—ñ—Å—Ç—å –ø–æ–≤—Ç–æ—Ä–µ–Ω—å, –∞–ª–µ –∑–Ω–∞—î–º–æ *—É–º–æ–≤—É –∑—É–ø–∏–Ω–∫–∏*.

**–ü—Ä–∏–∫–ª–∞–¥ (–°–∏–º—É–ª—è—Ç–æ—Ä –∑–±–æ—Ä—É –∫–æ—à—Ç—ñ–≤):**

```cpp

int main() {
    double totalGoal = 5000.0;
    double currentCollection = 0.0;
    
    std::cout << "–ü–æ—á–∏–Ω–∞—î–º–æ –∑–±—ñ—Ä –Ω–∞ " << totalGoal << " –≥—Ä–Ω." << std::endl;

    // 2. –£–º–æ–≤–∞: —Ü–∏–∫–ª –ø—Ä–∞—Ü—é—î, –ø–æ–∫–∏ –∑—ñ–±—Ä–∞–Ω–æ –º–µ–Ω—à–µ, –Ω—ñ–∂ –ø–æ—Ç—Ä—ñ–±–Ω–æ
    while (currentCollection < totalGoal) {

        // 3. –û–Ω–æ–≤–ª–µ–Ω–Ω—è: –¥–æ–¥–∞—î–º–æ –¥–æ–Ω–∞—Ç –¥–æ –∑–∞–≥–∞–ª—å–Ω–æ—ó —Å—É–º–∏
        currentCollection += newDonation;

    }

    std::cout << "–ó–±—ñ—Ä –∑–∞–≤–µ—Ä—à–µ–Ω–æ! –î—è–∫—É—î–º–æ!" << std::endl;

```

#### –¶–∏–∫–ª `for`

![–°—Ö–µ–º–∞ —Ä–æ–±–æ—Ç–∏ —Ü–∏–∫–ª—É for](attachments/02_for.png)

–¶–∏–∫–ª `for` —î –±—ñ–ª—å—à —Å—Ç—Ä—É–∫—Ç—É—Ä–æ–≤–∞–Ω–∏–º —ñ —á–∞—Å—Ç–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è, –∫–æ–ª–∏ –∫—ñ–ª—å–∫—ñ—Å—Ç—å –ø–æ–≤—Ç–æ—Ä–µ–Ω—å **–≤—ñ–¥–æ–º–∞ –∑–∞–∑–¥–∞–ª–µ–≥—ñ–¥—å**. –í—ñ–Ω –ø–æ—î–¥–Ω—É—î —ñ–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—é, —É–º–æ–≤—É —Ç–∞ –æ–Ω–æ–≤–ª–µ–Ω–Ω—è –≤ –æ–¥–Ω–æ–º—É —Ä—è–¥–∫—É.

**–°–∏–Ω—Ç–∞–∫—Å–∏—Å:** `for (—ñ–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—è; —É–º–æ–≤–∞; –æ–Ω–æ–≤–ª–µ–Ω–Ω—è) { ... }`

1.  **–Ü–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—è:** –í–∏–∫–æ–Ω—É—î—Ç—å—Å—è **–æ–¥–∏–Ω —Ä–∞–∑** –Ω–∞ —Å–∞–º–æ–º—É –ø–æ—á–∞—Ç–∫—É (–Ω–∞–ø—Ä., `int i = 0`).
2.  **–£–º–æ–≤–∞:** –ü–µ—Ä–µ–≤—ñ—Ä—è—î—Ç—å—Å—è **–ø–µ—Ä–µ–¥ –∫–æ–∂–Ω–æ—é —ñ—Ç–µ—Ä–∞—Ü—ñ—î—é** (–Ω–∞–ø—Ä., `i < 10`).
3.  **–û–Ω–æ–≤–ª–µ–Ω–Ω—è:** –í–∏–∫–æ–Ω—É—î—Ç—å—Å—è **–ø—ñ—Å–ª—è –∫–æ–∂–Ω–æ—ó —ñ—Ç–µ—Ä–∞—Ü—ñ—ó** (–Ω–∞–ø—Ä., `++i`).

![–î–µ—Ç–∞–ª—å–Ω–∞ —Å—Ö–µ–º–∞ —Ä–æ–±–æ—Ç–∏ —Ü–∏–∫–ª—É for](attachments/02_for_howto.png)

**–ü—Ä–∏–∫–ª–∞–¥ (–û–±—Ä–æ–±–∫–∞ 5 –ø–æ—Å–∏–ª–æ–∫):**

```cpp
#include <iostream>

int main() {
    int totalParcels = 5;
    std::cout << "–ü–æ—á–∏–Ω–∞—î–º–æ –æ–±—Ä–æ–±–∫—É –ø–æ—Å–∏–ª–æ–∫..." << std::endl;
    
    for (int i = 1; i <= totalParcels; ++i) {
        std::cout << "–û–±—Ä–æ–±–∫–∞ –ø–æ—Å–∏–ª–∫–∏ –Ω–æ–º–µ—Ä " << i << "..." << std::endl;
        // –¢—É—Ç –º—ñ–≥ –±–∏ –±—É—Ç–∏ —Å–∫–ª–∞–¥–Ω–∏–π –∫–æ–¥ –æ–±—Ä–æ–±–∫–∏
    }

    std::cout << "–í—Å—ñ –ø–æ—Å–∏–ª–∫–∏ –æ–±—Ä–æ–±–ª–µ–Ω–æ." << std::endl;

    return 0;
}
```

### –ö–µ—Ä—É–≤–∞–Ω–Ω—è —ñ—Ç–µ—Ä–∞—Ü—ñ—è–º–∏: `break` —Ç–∞ `continue`

–Ü–Ω–æ–¥—ñ –ø–æ—Ç—Ä—ñ–±–Ω–æ –∑–º—ñ–Ω–∏—Ç–∏ —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–∏–π –ø–æ—Ç—ñ–∫ –≤–∏–∫–æ–Ω–∞–Ω–Ω—è —Ü–∏–∫–ª—É.

  * **`break`**

      * **–©–æ —Ä–æ–±–∏—Ç—å:** –ù–µ–≥–∞–π–Ω–æ —ñ –ø–æ–≤–Ω—ñ—Å—Ç—é **–≤–∏—Ö–æ–¥–∏—Ç—å** –∑ –ø–æ—Ç–æ—á–Ω–æ–≥–æ (–Ω–∞–π–≥–ª–∏–±—à–æ–≥–æ) —Ü–∏–∫–ª—É.

  * **`continue`**

      * **–©–æ —Ä–æ–±–∏—Ç—å:** –ù–µ–≥–∞–π–Ω–æ **–ø—Ä–∏–ø–∏–Ω—è—î** –ø–æ—Ç–æ—á–Ω—É —ñ—Ç–µ—Ä–∞—Ü—ñ—é —ñ –ø–µ—Ä–µ—Ö–æ–¥–∏—Ç—å –¥–æ –Ω–∞—Å—Ç—É–ø–Ω–æ—ó. –ö–æ–¥ –≤ —Ç—ñ–ª—ñ —Ü–∏–∫–ª—É –ø—ñ—Å–ª—è `continue` –ø—Ä–æ–ø—É—Å–∫–∞—î—Ç—å—Å—è.

#### –ü—Ä–∏–∫–ª–∞–¥ —É –≤–∫–ª–∞–¥–µ–Ω–∏—Ö —Ü–∏–∫–ª–∞—Ö

**`break` —Ç–∞ `continue` –¥—ñ—é—Ç—å –ª–∏—à–µ –Ω–∞ —Ç–æ–π —Ü–∏–∫–ª, –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ —è–∫–æ–≥–æ –≤–æ–Ω–∏ –∑–Ω–∞—Ö–æ–¥—è—Ç—å—Å—è.**

```cpp
#include <iostream>

int main() {
    // –ó–æ–≤–Ω—ñ—à–Ω—ñ–π —Ü–∏–∫–ª (–Ω–∞–ø—Ä., 3 –∫–æ—Ä–æ–±–∫–∏)
    for (int i = 1; i <= 3; ++i) {
        std::cout << "–í—ñ–¥–∫—Ä–∏–≤–∞—î–º–æ –∫–æ—Ä–æ–±–∫—É " << i << std::endl;

        // –í–Ω—É—Ç—Ä—ñ—à–Ω—ñ–π —Ü–∏–∫–ª (–Ω–∞–ø—Ä., 5 –ø—Ä–µ–¥–º–µ—Ç—ñ–≤ —É –∫–æ—Ä–æ–±—Ü—ñ)
        for (int j = 1; j <= 5; ++j) {
            
            // –î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü—ñ—è continue: —ñ–≥–Ω–æ—Ä—É—î–º–æ "–Ω–µ–≤–∞–∂–ª–∏–≤—ñ" –ø—Ä–µ–¥–º–µ—Ç–∏
            if (j == 2) {
                std::cout << "  j=" << j << ": (–¥—Ä—ñ–±–Ω–∏—á–∫–∞, –ø—Ä–æ–ø—É—Å–∫–∞—î–º–æ)\n";
                continue; 
            }

            // –î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü—ñ—è break: –∑–Ω–∞–π—à–ª–∏ "—Ü—ñ–ª—å", —à—É–∫–∞—Ç–∏ –¥–∞–ª—ñ –≤ –¶–Ü–ô –∫–æ—Ä–æ–±—Ü—ñ –Ω–µ–º–∞—î —Å–µ–Ω—Å—É
            if (i * j > 6) {
                std::cout << "  i*j > 6 (" << i << "*" << j << "): –ó–Ω–∞–π—à–ª–∏ —Ç–µ, —â–æ —à—É–∫–∞–ª–∏! (break)\n";
                break; // –í–∏—Ö–æ–¥–∏—Ç—å –¢–Ü–õ–¨–ö–ò –∑ –≤–Ω—É—Ç—Ä—ñ—à–Ω—å–æ–≥–æ —Ü–∏–∫–ª—É
            }

            std::cout << "    –û–≥–ª—è–¥–∞—î–º–æ –ø—Ä–µ–¥–º–µ—Ç j = " << j << std::endl;
        }
        std::cout << "  ---> –ó–∞–∫—ñ–Ω—á–∏–ª–∏ –∑ –∫–æ—Ä–æ–±–∫–æ—é " << i << "\n";
    }
    std::cout << "--> –û–≥–ª—è–¥ –∑–∞–≤–µ—Ä—à–µ–Ω–æ.\n";
    return 0;
}
```

### ‚ùì –ü–∏—Ç–∞–Ω–Ω—è –¥–æ –≥—Ä—É–ø–∏

–£—è–≤—ñ—Ç—å, —â–æ –≤–∏ –ø–∏—à–µ—Ç–µ –∫–æ–¥ –¥–ª—è –±–∞–Ω–∫–æ–º–∞—Ç—É. –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å PIN-–∫–æ–¥. –£ –≤–∞—Å —î —Ü–∏–∫–ª `for`, —è–∫–∏–π –¥–∞—î 3 —Å–ø—Ä–æ–±–∏. –Ø–∫—â–æ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å –ø—Ä–∞–≤–∏–ª—å–Ω–∏–π PIN –Ω–∞ 2-–π —Å–ø—Ä–æ–±—ñ, —è–∫—É –∫–æ–º–∞–Ω–¥—É (`break` —á–∏ `continue`) –≤–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞—î—Ç–µ, —â–æ–± –∑—É–ø–∏–Ω–∏—Ç–∏ —Ü–∏–∫–ª? –ß–æ–º—É?

-----

### –¢—Ä–∏ –∫–∏—Ç–∏ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è: –ü–æ–≤–Ω–æ—Ç–∞ –∑–∞ –¢—å—é—Ä—ñ–Ω–≥–æ–º

–¢–µ–ø–µ—Ä, –∫–æ–ª–∏ –º–∏ –∑–Ω–∞—î–º–æ –ø—Ä–æ —Ç—Ä–∏ –æ—Å–Ω–æ–≤–Ω—ñ —Å—Ç—Ä—É–∫—Ç—É—Ä–∏ –∫–µ—Ä—É–≤–∞–Ω–Ω—è:

1.  **–ü–æ—Å–ª—ñ–¥–æ–≤–Ω–µ –≤–∏–∫–æ–Ω–∞–Ω–Ω—è** (–æ–¥–Ω–∞ –∫–æ–º–∞–Ω–¥–∞ –∑–∞ –æ–¥–Ω–æ—é)
2.  **–†–æ–∑–≥–∞–ª—É–∂–µ–Ω–Ω—è** (—É–º–æ–≤–∏ `if-else`)
3.  **–¶–∏–∫–ª–∏** (`while`, `for`)

–í–∞—Ä—Ç–æ –∑–Ω–∞—Ç–∏, —â–æ —Ü—å–æ–≥–æ –Ω–∞–±–æ—Ä—É **–¥–æ—Å—Ç–∞—Ç–Ω—å–æ –¥–ª—è —Ä–µ–∞–ª—ñ–∑–∞—Ü—ñ—ó –±—É–¥—å-—è–∫–æ–≥–æ, –Ω–∞—Å–∫—ñ–ª—å–∫–∏ –∑–∞–≤–≥–æ–¥–Ω–æ —Å–∫–ª–∞–¥–Ω–æ–≥–æ, –∞–ª–≥–æ—Ä–∏—Ç–º—É**. –ú–æ–≤–∞, —â–æ –º–∞—î —Ü—ñ —Ç—Ä–∏ —Å—Ç—Ä—É–∫—Ç—É—Ä–∏, –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è **–ø–æ–≤–Ω–æ—é –∑–∞ –¢—å—é—Ä—ñ–Ω–≥–æ–º**.

## `++x` –ø—Ä–æ—Ç–∏ `x++`: –£ —á–æ–º—É —Ä—ñ–∑–Ω–∏—Ü—è?

–¶–µ –∫–ª–∞—Å–∏—á–Ω–µ –ø–∏—Ç–∞–Ω–Ω—è –Ω–∞ —Å–ø—ñ–≤–±–µ—Å—ñ–¥–∞—Ö. –û–±–∏–¥–≤–∞ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∏ –∑–±—ñ–ª—å—à—É—é—Ç—å `x` –Ω–∞ –æ–¥–∏–Ω–∏—Ü—é, –∞–ª–µ **–ø–æ–≤–µ—Ä—Ç–∞—é—Ç—å** —Ä—ñ–∑–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è.

### `++x` (–ü—Ä–µ—Ñ—ñ–∫—Å–Ω–∏–π —ñ–Ω–∫—Ä–µ–º–µ–Ω—Ç)

**–ü—Ä–∞–≤–∏–ª–æ:** "–°–ø–æ—á–∞—Ç–∫—É –∑–±—ñ–ª—å—à–∏—Ç–∏, –ø–æ—Ç—ñ–º –ø–æ–≤–µ—Ä–Ω—É—Ç–∏".
–û–ø–µ—Ä–∞—Ç–æ—Ä —Å–ø–æ—á–∞—Ç–∫—É –∑–±—ñ–ª—å—à—É—î `x` –Ω–∞ 1, –∞ –ø–æ—Ç—ñ–º –ø–æ–≤–µ—Ä—Ç–∞—î **–Ω–æ–≤–µ, –≤–∂–µ –∑–±—ñ–ª—å—à–µ–Ω–µ** –∑–Ω–∞—á–µ–Ω–Ω—è.

```cpp
int x = 5;
int y = ++x; // –°–ø–æ—á–∞—Ç–∫—É x —Å—Ç–∞—î 6, –ø–æ—Ç—ñ–º —Ü–µ –∑–Ω–∞—á–µ–Ω–Ω—è 6 –ø—Ä–∏—Å–≤–æ—é—î—Ç—å—Å—è y
// –†–µ–∑—É–ª—å—Ç–∞—Ç: x = 6, y = 6
```

### `x++` (–ü–æ—Å—Ç—Ñ—ñ–∫—Å–Ω–∏–π —ñ–Ω–∫—Ä–µ–º–µ–Ω—Ç)

**–ü—Ä–∞–≤–∏–ª–æ:** "–°–ø–æ—á–∞—Ç–∫—É –ø–æ–≤–µ—Ä–Ω—É—Ç–∏, –ø–æ—Ç—ñ–º –∑–±—ñ–ª—å—à–∏—Ç–∏".
–û–ø–µ—Ä–∞—Ç–æ—Ä —Å–ø–æ—á–∞—Ç–∫—É –ø–æ–≤–µ—Ä—Ç–∞—î **–æ—Ä–∏–≥—ñ–Ω–∞–ª—å–Ω–µ** –∑–Ω–∞—á–µ–Ω–Ω—è `x`, —ñ –ª–∏—à–µ –ø—ñ—Å–ª—è —Ü—å–æ–≥–æ –∑–±—ñ–ª—å—à—É—î `x` –Ω–∞ 1.

```cpp
int x = 5;
int y = x++; // –°–ø–æ—á–∞—Ç–∫—É y –ø—Ä–∏—Å–≤–æ—é—î—Ç—å—Å—è –æ—Ä–∏–≥—ñ–Ω–∞–ª—å–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è x (5),
             // —ñ —Ç—ñ–ª—å–∫–∏ –ø—ñ—Å–ª—è —Ü—å–æ–≥–æ x —Å—Ç–∞—î 6
// –†–µ–∑—É–ª—å—Ç–∞—Ç: x = 6, y = 5
```
